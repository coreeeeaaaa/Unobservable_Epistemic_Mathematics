\section{Type Hierarchy}
\subsection{Observables (Material Objects)}

\begin{definition}[Scalar, Vector, Tensor]
Let $\mathcal{M}$ be the category of finite-dimensional real vector spaces with Euclidean topology. Material objects:
\begin{align*}
\mathsf{Scalar} &:= \mathbb{R}, \\
\mathsf{Vector} &:= \coprod_{n\in\mathbb{N}} \mathbb{R}^n, \\
\mathsf{Tensor}_k &:= \mathsf{Hom}((\mathbb{R}^n)^{\times k}, \mathbb{R}) \quad (k\ge 2).
\end{align*}
\end{definition}

\begin{definition}[Observable Object]
\[
\mathsf{ObsObject} := \mathsf{Scalar} \uplus \mathsf{Vector} \uplus \coprod_{k\ge 2} \mathsf{Tensor}_k
\]
\end{definition}

\subsection{Epistemics (Unobservable Objects)}

\begin{definition}[World and Observer]
Let $X$ be Polish with Borel $\sigma$-algebra $\mathcal{B}(X)$ and measure $\mu$.
\begin{align*}
\mathsf{World} &:= \mathcal{P}(X), \\
\mathsf{Observer} &:= \{ \Pi: X \to \mathsf{ObsObject} \mid \Pi \text{ Borel} \}, \\
M_O(w) &:= \{ x\in w \mid \exists y\in w,\, \Pi_O(x)=\Pi_O(y),\, x\neq y \}, \\
\tau_O(w) &:= \mu(M_O(w)).
\end{align*}
\end{definition}

\begin{definition}[Epistemic Objects]
\begin{align*}
\mathsf{Spark} &:= \{ \delta_x \mid x\in X \}, \\
\mathsf{Actyon} &:= \mathsf{World} \times \mathsf{World}^{op} \quad (\text{Intent-bearing structures}), \\
\mathsf{Escalade} &:= \{ f: \mathsf{World} \to \mathsf{World} \mid f \text{ is contractive w.r.t. } \tau \}, \\
\mathsf{Secare} &:= \{ w \in \mathsf{World} \mid \forall \Pi \in \mathsf{Observer}, \mu(M_\Pi(w)) = 0 \}.
\end{align*}
\end{definition}

\begin{definition}[Unobservable Object]
\[
\mathsf{UnobsObject} := \mathsf{Spark} \uplus \mathsf{Actyon} \uplus \mathsf{Escalade} \uplus \mathsf{Secare}
\]
\end{definition}

\begin{definition}[Complete Object Hierarchy]
\[
\mathsf{UEMObject} := \mathsf{ObsObject} \uplus \mathsf{UnobsObject} \uplus \mathsf{Observer} \uplus \mathsf{World}
\]
\end{definition}
